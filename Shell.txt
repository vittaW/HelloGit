# CMD #
cd xxx 进入某个路径
cd ... 返回上级路径
dir    查看当前目录
mkdir  创建文件夹

# Git #
git clone xxx.git   下载一个Git仓库
git status	     查看git代码状态
git add xxx	     将某个文件添加到git中
git add -A 	     将所有文件添加到git中
git commit           提交到本地仓库
git commit -m "xxx"  提交到本地仓库,并添加描述
git pull 	     更新代码,(先更新再提交),从远程仓库更新
git push	     提交到远程仓库
